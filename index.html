A data grid for AngularJS

  • Supports different number of columns

  • Column widths are equal if not specified manually.

	var tableConfig = [{
		name: 'Id',
		field: 'id',
		defaultSearchForMobile: true,
		showSearch: false
	}, {
		name: 'First Name',
		field: 'first_name',
		showSearch: true
	}, {
		name: 'Last Name',
		field: 'last_name',
		showSearch: true
	}];

	$rootScope.threeColTableConf = {
		data: tableData,
		safeData: safeTableData,
		config: tableConfig,
		tableBodyHeight: 250,
		showActions: true,
		actionsConfig: {
			title: 'Actions',
			edit: {
				stateName: 'editrole',
				routerUrlParam: 'roleId',
				fieldName: 'id'
			},
			view: {
				stateName: 'viewrole',
				routerUrlParam: 'roleId',
				fieldName: 'id'
			},
			customWidthApplied: '10%'
		}
	};

  • You can remove 'Actions'column and/or rename it.

  • You can also disable search fields if needed. See 'ID' column.

	var tableConfig = [{
		name: 'Id',
		field: 'id',
		defaultSearchForMobile: true,
		showSearch: false
	}, {
		name: 'First Name',
		field: 'first_name',
		showSearch: true
	}, {
		name: 'Last Name',
		field: 'last_name',
		showSearch: true
	}];

	$rootScope.withoutActionsColConf = {
		data: tableData,
		safeData: safeTableData,
		config: tableConfig,
		tableBodyHeight: 250
	};

Dynamically set column width

Columns are now 15%/30%/55%

	var tableConfig = [{
		name: 'Id',
		field: 'id',
		width: '15%',
		showSearch: true
	}, {
		name: 'First Name',
		field: 'first_name',
		width: '30%',
		showSearch: true
	}, {
		name: 'Last Name',
		field: 'last_name',
		width: '55%',
		showSearch: true
	}];

	$rootScope.predefinedColWidthConf = {
		data: tableData,
		safeData: safeTableData,
		config: tableConfig,
		tableBodyHeight: 200
	};

Remove search fields if needed.

There are 2 properties, one for each (desktop or mobile) view, so you can choose to leave one or another. Or leave fields for both views.

This example shows hidden inputs on both mobile and desktop views.

	var tableConfig = [{
		name: 'Id',
		field: 'id',
		showSearch: true
	}, {
		name: 'First Name',
		field: 'first_name',
		showSearch: true
	}, {
		name: 'Last Name',
		field: 'last_name',
		showSearch: true
	}];
	$rootScope.noSearchFields = {
		data: tableData,
		safeData: safeTableData,
		config: tableConfig,
		hideSearchInputsDesk: true,
		hideSearchInputsMobile: true,
		tableBodyHeight: 200
	};
